-- UOM Table
DROP TABLE dbo.TNtlUom;

CREATE TABLE dbo.TNtlUom(
    id INT IDENTITY(1, 1) not null,
    name VARCHAR(50),
    detail_id INT,
    CONSTRAINT uom_id_pk PRIMARY KEY(id)
);

-- Status Table
DROP TABLE dbo.TNtlStatus;

CREATE TABLE dbo.TNtlStatus(
    id INT IDENTITY(1, 1) not null,
    name VARCHAR(50),
    detail_id INT,
    CONSTRAINT status_id_pk PRIMARY KEY(id)
);
